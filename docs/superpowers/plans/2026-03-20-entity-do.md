# Entity DO Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a lightweight Durable Object class (`EntityDB`) that holds a single entity instance with JSON data, versioning, and CDC event logging.

**Architecture:** Each Entity DO holds one row in an `entity` table and an append-only `events` table for CDC. RPC methods provide CRUD. The entity table and events table share the same SQLite transaction for atomicity. This is Plan 1 of 4 for the DO-per-entity architecture.

**Tech Stack:** TypeScript, Cloudflare Workers Durable Objects, SQLite, Vitest

**Spec:** `docs/superpowers/specs/2026-03-20-do-per-entity-design.md`

// src/entity-do.ts

/**
 * EntityDB — a single entity instance as a Durable Object.
 *
 * Each DO holds one row in `entity` + an append-only `events` table for CDC.
 * SQLite is used (not KV) because the events table benefits from SQL queries
 * and both tables share the same transaction for atomicity.
 *
 * This file exports pure functions that operate on a SqlStorage-like interface.
 * The actual DO class (EntityDB) is a thin wrapper in index.ts.
 */

export interface EntityData {
  id: string
  type: string
  data: Record<string, unknown>
}

export interface EntityRecord {
  id: string
  type: string
  data: string  // JSON
  version: number
  created_at: string
  updated_at: string
  deleted_at: string | null
}

export interface EventRecord {
  id: string
  timestamp: string
  operation: 'create' | 'update' | 'delete'
  data: string | null   // JSON: current state
  prev: string | null   // JSON: previous state
}

export interface SqlLike {
  exec(query: string, ...params: any[]): { toArray(): any[] }
}

export function initEntitySchema(sql: SqlLike): void {
  sql.exec(`CREATE TABLE IF NOT EXISTS entity (
    id TEXT PRIMARY KEY,
    type TEXT NOT NULL,
    data TEXT NOT NULL,
    version INTEGER NOT NULL DEFAULT 1,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now')),
    deleted_at TEXT
  )`)
  sql.exec(`CREATE TABLE IF NOT EXISTS events (
    id TEXT PRIMARY KEY,
    timestamp TEXT NOT NULL DEFAULT (datetime('now')),
    operation TEXT NOT NULL,
    data TEXT,
    prev TEXT
  )`)
}

function logEvent(sql: SqlLike, operation: string, data: string | null, prev: string | null): void {
  const id = crypto.randomUUID()
  const now = new Date().toISOString()
  sql.exec(
    'INSERT INTO events (id, timestamp, operation, data, prev) VALUES (?, ?, ?, ?, ?)',
    id, now, operation, data, prev,
  )
}

export function createEntity(sql: SqlLike, input: EntityData): { id: string; version: number } {
  const now = new Date().toISOString()
  const dataJson = JSON.stringify(input.data)
  sql.exec(
    'INSERT INTO entity (id, type, data, version, created_at, updated_at) VALUES (?, ?, ?, 1, ?, ?)',
    input.id, input.type, dataJson, now, now,
  )
  logEvent(sql, 'create', dataJson, null)
  return { id: input.id, version: 1 }
}
```

// append to entity-do.ts
export function updateEntity(sql: SqlLike, fields: Record<string, unknown>): { id: string; version: number } | null {
  const rows = sql.exec('SELECT * FROM entity LIMIT 1').toArray()
  if (!rows.length) return null
  const row = rows[0] as EntityRecord
  const prevData = row.data
  const currentData = JSON.parse(prevData)
  const newData = { ...currentData, ...fields }
  const newDataJson = JSON.stringify(newData)
  const newVersion = row.version + 1
  const now = new Date().toISOString()
  sql.exec(
    'UPDATE entity SET data = ?, version = ?, updated_at = ? WHERE id = ?',
    newDataJson, newVersion, now, row.id,
  )
  logEvent(sql, 'update', newDataJson, prevData)
  return { id: row.id, version: newVersion }
}

export function getEvents(sql: SqlLike, since?: string): EventRecord[] {
  if (since) {
    return sql.exec(
      'SELECT * FROM events WHERE timestamp > ? ORDER BY timestamp DESC', since,
    ).toArray() as EventRecord[]
  }
  return sql.exec('SELECT * FROM events ORDER BY timestamp DESC').toArray() as EventRecord[]
}

// append to entity-do.ts
export function deleteEntity(sql: SqlLike): { id: string; deleted: boolean } | null {
  const rows = sql.exec('SELECT * FROM entity LIMIT 1').toArray()
  if (!rows.length) return null
  const row = rows[0] as EntityRecord
  const now = new Date().toISOString()
  sql.exec('UPDATE entity SET deleted_at = ?, updated_at = ? WHERE id = ?', now, now, row.id)
  logEvent(sql, 'delete', null, row.data)
  return { id: row.id, deleted: true }
}

// append to entity-do.ts
import { DurableObject } from 'cloudflare:workers'

export class EntityDB extends DurableObject {
  private initialized = false

  private ensureInit(): void {
    if (this.initialized) return
    initEntitySchema(this.ctx.storage.sql)
    this.initialized = true
  }

  async get(): Promise<ReturnType<typeof getEntity>> {
    this.ensureInit()
    return getEntity(this.ctx.storage.sql)
  }

  async put(input: EntityData): Promise<{ id: string; version: number }> {
    this.ensureInit()
    const existing = getEntity(this.ctx.storage.sql)
    if (existing) {
      return updateEntity(this.ctx.storage.sql, input.data as Record<string, unknown>)!
    }
    return createEntity(this.ctx.storage.sql, input)
  }

  async patch(fields: Record<string, unknown>): Promise<{ id: string; version: number } | null> {
    this.ensureInit()
    return updateEntity(this.ctx.storage.sql, fields)
  }

  async delete(): Promise<{ id: string; deleted: boolean } | null> {
    this.ensureInit()
    return deleteEntity(this.ctx.storage.sql)
  }

  async events(since?: string): Promise<EventRecord[]> {
    this.ensureInit()
    return getEvents(this.ctx.storage.sql, since)
  }
}
```
